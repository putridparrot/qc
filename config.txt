# Use 0 to disable history entirely.

# Safety policy for dangerous commands:
# warn    = print warning but continue
# confirm = prompt before executing dangerous commands
# block   = block dangerous commands unless command contains --force

# Dry-run mode:
# true  = preview and validate commands but do not execute
# false = execute normally
dry_run=false

# Active shortcut profile:
# default uses shortcuts.txt
# other values use shortcuts.<profile>.txt
active_profile=default

# Prompt display for default profile:
# false = show "QC <cwd>" when profile is default
# true  = show "QC [default] <cwd>"
show_default_profile_in_prompt=false
